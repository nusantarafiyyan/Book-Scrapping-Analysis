BASE_URL = "https://books.toscrape.com/"
headers = {"User-Agent": "Mozilla/5.0"}

def get_soup(url):
    res = requests.get(url, headers=headers)
    if res.status_code == 404:
        return None          # sinyal: halaman tidak ada
    res.raise_for_status()
    return BeautifulSoup(res.text, "html.parser")

def parse_book(card):
    a_tag = card.h3.a
    title = a_tag["title"].strip()

    price_tag = card.find("p", class_="price_color")
    price_text = price_tag.get_text(strip=True) if price_tag else ""
    clean = (
        price_text.replace("Â", "")
                  .replace("£", "")
                  .strip()
    )
    try:
        price = float(clean) if clean else None
    except ValueError:
        price = None

    rating_tag = card.find("p", class_="star-rating")
    rating = None
    if rating_tag:
        classes = rating_tag.get("class", [])
        rating_map = {"One": 1, "Two": 2, "Three": 3, "Four": 4, "Five": 5}
        for c in classes:
            if c in rating_map:
                rating = rating_map[c]

    stock_tag = card.find("p", class_="instock availability")
    stock_text = stock_tag.get_text(strip=True) if stock_tag else ""

    link_rel = a_tag["href"]
    if link_rel.startswith("../"):
        link_rel = link_rel.replace("../", "", 1)
    book_url = BASE_URL + link_rel

    return {
        "title": title,
        "price": price,
        "rating": rating,
        "stock_text": stock_text,
        "url": book_url,
    }

def scrape_all_books():
    all_books = []
    page = 1
    while True:
        if page == 1:
            url = BASE_URL
        else:
            url = f"{BASE_URL}catalogue/page-{page}.html"

        print("Scraping:", url)
        soup = get_soup(url)
        if soup is None:
            print("Halaman 404, stop.")
            break

        cards = soup.find_all("article", class_="product_pod")
        if not cards:
            print("Tidak ada buku lagi, stop di halaman", page)
            break

        for card in cards:
            all_books.append(parse_book(card))

        page += 1
        time.sleep(1)

    return all_books

# ==============================
# 1️⃣ Shapiro-Wilk Test
# ==============================
shapiro_stat, shapiro_p = stats.shapiro(data)

print("=== Shapiro-Wilk Test ===")
print(f"Statistic: {shapiro_stat:.4f}")
print(f"P-value  : {shapiro_p:.6f}")

if shapiro_p > 0.05:
    print("Kesimpulan: Data berdistribusi normal")
else:
    print("Kesimpulan: Data TIDAK berdistribusi normal")


# ==============================
# 2️⃣ Kolmogorov-Smirnov Test
# ==============================

# Standarisasi data dulu
data_std = (data - np.mean(data)) / np.std(data)

ks_stat, ks_p = stats.kstest(data_std, 'norm')

print("\n=== Kolmogorov-Smirnov Test ===")
print(f"Statistic: {ks_stat:.4f}")
print(f"P-value  : {ks_p:.6f}")

if ks_p > 0.05:
    print("Kesimpulan: Data berdistribusi normal")
else:
    print("Kesimpulan: Data TIDAK berdistribusi normal")
